<script>

function hesapla() {

    const rgbInput = document.getElementById('rgb').value.trim();

    const totalAmount = parseFloat(document.getElementById('amount').value);

    if(!/^#([0-9A-Fa-f]{6})$/.test(rgbInput) || isNaN(totalAmount) || totalAmount <= 0) {

        alert('Lütfen geçerli bir RGB kodu ve miktar girin!');

        return;

    }

    const r = parseInt(rgbInput.substr(1,2), 16);

    const g = parseInt(rgbInput.substr(3,2), 16);

    const b = parseInt(rgbInput.substr(5,2), 16);

    document.getElementById('colorPreview').style.background = rgbInput;

    let total = r + g + b;

    if(total === 0) total = 1; // Siyah için sıfıra bölmeyi engelle

    const rRatio = r / total;

    const gRatio = g / total;

    const bRatio = b / total;

    // Şeffaflık oranı

    const transparentRatio = (3 - ((r/255)+(g/255)+(b/255))) / 3;

    const ratioSum = rRatio + gRatio + bRatio + transparentRatio;

    const rWeight = (rRatio / ratioSum) * totalAmount;

    const gWeight = (gRatio / ratioSum) * totalAmount;

    const bWeight = (bRatio / ratioSum) * totalAmount;

    const tWeight = (transparentRatio / ratioSum) * totalAmount;

    // Sıvı miktarı: %20

    const liquid = totalAmount * 0.20;

    document.getElementById('result').innerHTML = `

        <strong>Sonuç:</strong><br>

        Kırmızı: ${rWeight.toFixed(2)} g<br>

        Yeşil: ${gWeight.toFixed(2)} g<br>

        Mavi: ${bWeight.toFixed(2)} g<br>

        Şeffaf: ${tWeight.toFixed(2)} g<br>

        Sıvı: ${liquid.toFixed(2)} ml

    `;

}

</script>
